# solution
library(tidyverse)
options(repr.plot.width = 5, repr.plot.height = 3)
plot <- ggplot(iris,aes(x = Petal.Length, y = Petal.Width)) + 
    geom_point(aes(color = Species, shape = Species)) +
    xlab("Petal Length") +
    ylab("Petal Width")
